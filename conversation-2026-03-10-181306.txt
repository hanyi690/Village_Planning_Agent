   class CheckpointType(str, Enum):
       """检查点类型"""
       KEY = "key"           # 关键检查点：层级/维度完成，用户可回滚
       REGULAR = "regular"   # 普通检查点：LangGraph 自动保存的中间节点

   class PlanningPhase(str, Enum):
       """规划阶段枚举"""
       INIT = "init"
       LAYER1_ANALYZING = "layer1_analyzing"
       LAYER1_COMPLETED = "layer1_completed"
       LAYER2_CONCEPTING = "layer2_concepting"
       LAYER2_COMPLETED = "layer2_completed"
       LAYER3_PLANNING = "layer3_planning"
       LAYER3_COMPLETED = "layer3_completed"
       FINAL_OUTPUT = "final_output"

   class CheckpointMetadata(BaseModel):
       """检查点元数据"""
       type: CheckpointType = CheckpointType.REGULAR
       phase: PlanningPhase = PlanningPhase.INIT
       layer: int = 0                      # 所属层级
       dimension_key: Optional[str] = None  # 维度 key（维度级检查点）
       dimension_name: Optional[str] = None # 维度名称（用于展示）
       description: str = ""

       def to_dict(self) -> dict:
           return self.model_dump()

       @classmethod
       def from_dict(cls, data: dict) -> "CheckpointMetadata":
           if isinstance(data.get("type"), str):
               data["type"] = CheckpointType(data["type"])
           if isinstance(data.get("phase"), str):
               data["phase"] = PlanningPhase(data["phase"])
           return cls(**data)
