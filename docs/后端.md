#### 启动规划

```python
@router.post("/start")
async def start_planning(request: StartPlanningRequest) -> StartPlanningResponse:
    """
    启动村庄规划任务

    流程:
    1. 验证输入数据
    2. 创建会话 ID
    3. 初始化 LangGraph 状态
    4. 返回 session_id 用于 SSE 连接
    """
    session_id = str(uuid.uuid4())

    # 初始化状态
    initial_state = {
        "project_name": request.project_name,
        "village_data": request.village_data,
        "task_description": request.task_description,
        "constraints": request.constraints,
        "current_layer": 1,
        "step_mode": request.step_mode,
        "enable_review": request.enable_review,
    }

    # 保存会话状态
    sessions[session_id] = {
        "state": initial_state,
        "status": "pending",
        "created_at": datetime.now(),
    }

    return {
        "success": True,
        "session_id": session_id,
        "status": "pending",
        "message": "规划任务已启动"
    }
```

#### SSE 流式传输

```python
@router.get("/stream/{session_id}")
async def stream_planning(
    session_id: str,
    request: Request
) -> StreamingResponse:
    """
    SSE 流式传输规划进度

    事件类型:
    - progress: 进度更新
    - layer_completed: 层级完成
    - pause: 暂停等待审查
    - stream_paused: 流已暂停,前端应关闭连接 ⭐
    - error: 错误消息
    - completed: 任务完成
    """

    async def event_generator():
        try:
            # 获取会话状态
            session = sessions.get(session_id)
            if not session:
                yield format_sse("error", {"message": "会话不存在"})
                return

            # 执行规划图
            graph = create_village_planning_graph()

            async for event in graph.astream(
                session["state"],
                config={"configurable": {"thread_id": session_id}}
            ):
                # 检查客户端断开
                if await request.is_disconnected():
                    logger.info(f"[SSE] Client {session_id} disconnected")
                    break

                # 解析事件并发送
                if "pause_after_step" in event:
                    # 暂停状态
                    yield format_sse("pause", {
                        "session_id": session_id,
                        "current_layer": event["current_layer"],
                        "reason": "pause_after_step"
                    })

                    # ⭐ 通知前端关闭连接
                    yield format_sse("stream_paused", {
                        "session_id": session_id,
                        "message": "Stream paused, please close connection"
                    })

                    # 设置流状态
                    _stream_states[session_id] = "paused"
                    break

                # 其他事件处理...
                yield format_sse("progress", event)

        except Exception as e:
            logger.error(f"[SSE] Error in stream: {e}")
            yield format_sse("error", {"message": str(e)})

    return StreamingResponse(
        event_generator(),
        media_type="text/event-stream",
        headers={
            "Cache-Control": "no-cache",
            "Connection": "keep-alive",
        }
    )
```

#### 审查操作

```python
@router.post("/review/{session_id}")
async def handle_review(
    session_id: str,
    request: ReviewActionRequest
) -> ReviewActionResponse:
    """
    处理审查操作

    操作类型:
    - approve: 批准继续
    - reject: 驳回修改
    - rollback: 回退到检查点
    """
    session = sessions.get(session_id)
    if not session:
        raise HTTPException(404, "会话不存在")

    action = request.action

    if action == "approve":
        # 批准,恢复执行
        result = await approve_review(session_id, request.feedback)
        return {
            "success": True,
            "message": "已批准,继续执行",
            "resumed": True,
            "current_layer": result.get("current_layer")
        }

    elif action == "reject":
        # 驳回,触发修改
        result = await reject_review(
            session_id,
            request.feedback,
            request.dimensions
        )
        return {
            "success": True,
            "message": "已驳回,触发修改",
            "resumed": False
        }

    elif action == "rollback":
        # 回退到检查点
        result = await rollback_to_checkpoint(
            session_id,
            request.checkpoint_id
        )
        return {
            "success": True,
            "message": f"已回退到 {request.checkpoint_id}",
            "resumed": False
        }

    else:
        raise HTTPException(400, f"未知操作: {action}")
```

#### 回退到检查点

```python
@router.post("/rollback/{session_id}")
async def rollback_to_checkpoint(
    session_id: str,
    request: RollbackRequest
) -> RollbackResponse:
    """
    回退到指定检查点并恢复执行

    流程:
    1. 加载检查点状态
    2. 更新 current_layer
    3. 清除 pause_after_step 标志
    4. 从检查点恢复执行
    """
    session = sessions.get(session_id)
    if not session:
        raise HTTPException(404, "会话不存在")

    # 加载检查点
    checkpoint = checkpoint_manager.load(request.checkpoint_id)
    if not checkpoint["success"]:
        raise HTTPException(404, "检查点不存在")

    # 更新会话状态
    session["state"] = checkpoint["state"]
    session["state"]["current_layer"] = checkpoint["layer"]
    session["state"]["pause_after_step"] = False

    return {
        "success": True,
        "message": f"已回退到 {request.checkpoint_id}",
        "current_layer": checkpoint["layer"]
    }
```

```python
@router.post("/upload")
async def upload_file(
    file: UploadFile,
    encoding: Optional[str] = None
) -> UploadResponse:
    """
    文件上传和解析

    支持格式:
    - .txt: 纯文本,自动检测编码
    - .md: Markdown,自动检测编码
    - .docx: Word 文档
    - .pdf: PDF 文档

    编码检测:
    - UTF-8
    - GBK
    - GB2312
    """
    try:
        # 读取文件内容
        content = await file.read()

        # 根据文件类型解析
        if file.filename.endswith(('.txt', '.md')):
            # 文本文件 - 编码检测
            text, detected_encoding = detect_encoding(content)
            file_content = text.decode(detected_encoding)

        elif file.filename.endswith('.docx'):
            # Word 文档
            file_content = parse_docx(content)

        elif file.filename.endswith('.pdf'):
            # PDF 文档
            file_content = parse_pdf(content)

        else:
            raise ValueError(f"不支持的文件类型: {file.filename}")

        # 验证内容
        if len(file_content) < 10:
            raise ValueError("文件内容太短")

        return {
            "success": True,
            "filename": file.filename,
            "encoding": detected_encoding,
            "content": file_content,
            "content_length": len(file_content)
        }

    except Exception as e:
        logger.error(f"[File Upload] Error: {e}")
        raise HTTPException(500, str(e))
```

#### 编码检测

```python
def detect_encoding(content: bytes) -> Tuple[bytes, str]:
    """
    自动检测文本编码

    尝试顺序:
    1. UTF-8 (最常见)
    2. GBK (中文 Windows)
    3. GB2312 (简体中文)
    """
    encodings = ['utf-8', 'gbk', 'gb2312']

    for encoding in encodings:
        try:
            decoded = content.decode(encoding)
            # 验证解码成功
            if decoded and len(decoded) > 0:
                return content, encoding
        except UnicodeDecodeError:
            continue

    # 如果都失败,使用 UTF-8 并忽略错误
    return content, 'utf-8'
```

@router.get("/sessions")
async def get_sessions() -> SessionsResponse:
    """
    获取所有会话列表
    """
    session_list = []

    for session_id, session_data in sessions.items():
        session_list.append({
            "session_id": session_id,

        "count": len(session_list)
    }

        "can_resume": True

#### 核心方法

```python
class PlanningService:
    """规划服务"""

    def __init__(self):
        self.graph = None
        self.checkpointer = None

    def create_graph(self) -> StateGraph:
        """
        创建村庄规划图

        返回配置好的 LangGraph StateGraph
        """
        from src.orchestration.main_graph import create_village_planning_graph

        self.graph = create_village_planning_graph()
        return self.graph

    def execute_layer(
        self,
        session_id: str,
        state: Dict[str, Any],
        layer: int
    ) -> Dict[str, Any]:
        """
        执行指定层级

        Args:
            session_id: 会话 ID
            state: 当前状态
            layer: 层级编号 (1, 2, 3)

        Returns:
            更新后的状态
        """
        if not self.graph:
            self.create_graph()

        # 更新 current_layer
        state["current_layer"] = layer

        # 执行图
        config = {"configurable": {"thread_id": session_id}}
        result = self.graph.invoke(state, config)

        return result

    def filter_state_for_dimension(
        self,
        state: Dict[str, Any],
        dimension_key: str
    ) -> Dict[str, Any]:
        """
        为特定维度筛选状态

        只传递该维度依赖的相关数据
        """
        from src.utils.state_filter import filter_state_for_dimension

        return filter_state_for_dimension(state, dimension_key)
```

class StartPlanningRequest(BaseModel):
