import type { Message, SSEEvent } from './message';
import type { ProgressMessage, ActionMessage } from './message-types';

/**
 * 将 SSE 事件转换为前端消息
 */
export function convertSSEToMessage(event: SSEEvent): Message {
  const base = {
    id: `msg-${Date.now()}-${Math.random()}`,
    timestamp: new Date(),
    role: 'assistant' as const,
  };

  // 进度更新
  if (event.status === 'running' && event.progress !== undefined) {
    return {
      ...base,
      type: 'progress',
      content: event.message || '正在处理...',
      progress: event.progress,
      currentLayer: event.current_layer,
      taskId: event.task_id,
    } as ProgressMessage;
  }

  // 暂停状态
  if (event.status === 'paused' || event.event_type === 'pause') {
    return {
      ...base,
      type: 'action',
      content: '当前层级已完成，请审查后继续',
      actions: createActionButtons('pause'),
      taskId: event.task_id,
    } as ActionMessage;
  }

  // ... 其他转换逻辑
}

1. **接口统一** - 合并 `UseTaskSSEOptions` 和 `SSEHandlerCallbacks`
2. **事件映射** - 使用 `EVENT_HANDLERS` 对象替代复杂 switch

#### 使用示例

```typescript
const { isConnected, error, close, reconnect } = useTaskSSE(taskId, {
  onProgress: (data) => {
    console.log('进度更新:', data.progress);
  },

  onLayerCompleted: (data) => {
    console.log('层级完成:', data.layer);
    // 显示层级报告

  },

  onPause: (data) => {
    console.log('任务暂停:', data.reason);
    // 显示审查面板

  },

  onStreamPaused: () => {
    // 前端主动关闭连接

  },

  onError: (error) => {
    console.error('SSE 错误:', error);
    // 显示错误消息
    showErrorMessage(error);
  }
});

// 批准后手动重连
const handleApprove = async () => {
  await planningApi.approveReview(taskId);
  reconnect();  // 创建新的 SSE 连接
};
```

#### 事件处理器映射 ⭐ NEW

```typescript
// 简化的事件处理
const EVENT_HANDLERS = {
  progress: (e: MessageEvent) => {
    const data = JSON.parse(e.data);
    onProgress?.(data);
  },

  layer_completed: (e: MessageEvent) => {
    const data = JSON.parse(e.data);
    onLayerCompleted?.(data);
  },

  pause: (e: MessageEvent) => {
    const data = JSON.parse(e.data);
    onPause?.(data);
  },

  stream_paused: (e: MessageEvent) => {
    console.log('[SSE] Stream paused, closing connection');
    eventSourceRef.current?.close();
    onStreamPaused?.();
  },

  error: (e: MessageEvent) => {
    const data = JSON.parse(e.data);
    onError?.(data.message || 'SSE error');
  },

  completed: (e: MessageEvent) => {
    onCompleted?.();
  }
};

// 注册事件监听器
Object.entries(EVENT_HANDLERS).forEach(([event, handler]) => {
  es.addEventListener(event, handler);
});
```

#### SSE 连接生命周期

```
1. 创建连接
   useTaskSSE(taskId)
      ↓
   创建 EventSource
      ↓
   GET /api/planning/stream/{session_id}

2. 接收事件
   progress → 更新进度
   layer_completed → 显示完成消息
   pause → 显示审查面板
   stream_paused → 关闭连接 ⭐
   error → 显示错误

3. 暂停状态
   后端发送 stream_paused
      ↓
   前端关闭 EventSource
      ↓
   连接状态 → disconnected

4. 批准后重连
   用户点击批准
      ↓
   调用 reconnect()
      ↓
   重置初始化标志
      ↓
   创建新的 EventSource
      ↓
   继续执行

5. 完成状态
   后端发送 completed
      ↓
   前端关闭 EventSource
      ↓
   连接状态 → completed
```

#### API 方法

```typescript
const planningApi = {
  // 启动规划
  startPlanning: async (data: StartPlanningRequest): Promise<StartPlanningResponse> => {
    const response = await fetch(`${API_URL}/planning/start`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(data),
    });
    return response.json();
  },

  // 文件上传
  uploadFile: async (file: File, encoding?: string): Promise<UploadResponse> => {
    const formData = new FormData();
    formData.append('file', file);
    if (encoding) {
      formData.append('encoding', encoding);
    }

    const response = await fetch(`${API_URL}/files/upload`, {
      method: 'POST',
      body: formData,
    });
    return response.json();
  },

  // 审查操作
  approveReview: async (taskId: string): Promise<ReviewResponse> => {
    const response = await fetch(`${API_URL}/planning/review/${taskId}`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ action: 'approve' }),
    });
    return response.json();
  },

  rejectReview: async (
    taskId: string,
    feedback: string,
    dimensions?: string[]
  ): Promise<ReviewResponse> => {
    const response = await fetch(`${API_URL}/planning/review/${taskId}`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        action: 'reject',
        feedback,
        dimensions,
      }),
    });
    return response.json();
  },

  // 回退到检查点
  rollbackToCheckpoint: async (
    taskId: string,
    checkpointId: string
  ): Promise<RollbackResponse> => {
    const response = await fetch(`${API_URL}/planning/rollback/${taskId}`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ checkpoint_id: checkpointId }),
    });
    return response.json();
  },

  // 获取会话列表
  getSessions: async (): Promise<SessionListResponse> => {
    const response = await fetch(`${API_URL}/data/sessions`);
    return response.json();
  },

  // 获取会话详情
  getSession: async (sessionId: string): Promise<SessionDetailResponse> => {
    const response = await fetch(`${API_URL}/data/sessions/${sessionId}`);
    return response.json();
  },
};

export default planningApi;
```

## 常见问题

### Q: 如何添加新的消息类型?

A:
1. 在 `message-types.ts` 中定义接口
2. 在 `message.ts` 的 `Message` 联合类型中添加
3. 在 `message-guards.ts` 中添加类型守卫
4. 在 `MessageContent.tsx` 中添加渲染逻辑

### Q: 如何自定义 SSE 事件处理?

A: 在 `useTaskSSE` 中传入自定义回调:
```typescript
useTaskSSE(taskId, {
  onCustomEvent: (data) => { /* 处理 */ }
});
```

### Q: 如何调试消息流?

A: 使用 Redux DevTools 或在 Context 中添加日志:
```typescript
addMessage: (message) => {
  console.log('[Context] Adding message:', message);
  setMessages(prev => [...prev, message]);
}
```
