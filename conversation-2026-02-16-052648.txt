   def generate_wave_config_from_dependencies() -> Dict[int, List[str]]:
       """根据依赖关系自动生成 Wave 配置"""
       dims = list_detailed_dimensions()
       wave_config = {}
       assigned = set()
       wave = 1

       while len(assigned) < len(dims):
           # 找出当前 Wave 可以执行的维度（所有依赖已满足）
           ready_dims = []
           for dim in dims:
               if dim["key"] in assigned:
                   continue
               deps = dim.get("dependencies", {}).get("layer3_plans", [])
               if all(d in assigned for d in deps):
                   ready_dims.append(dim["key"])

           if ready_dims:
               wave_config[wave] = ready_dims
               assigned.update(ready_dims)
               wave += 1
           else:
               break  # 避免死循环

       return wave_config

   class IncrementalExecutor:
       """增量执行器 - 支持只执行失败的维度"""

       def __init__(self, checkpoint_manager, state_filter):
           self.checkpoint_manager = checkpoint_manager
           self.state_filter = state_filter

       def get_pending_dimensions(
           self,
           all_dimensions: List[str],
           completed: Dict[str, str],
           failed: List[str]
       ) -> List[str]:
           """获取待执行的维度（排除已完成且成功的）"""
           pending = []
           for dim in all_dimensions:
               if dim not in completed or dim in failed:
                   pending.append(dim)
           return pending

       def resume_from_checkpoint(
           self,
           checkpoint_id: str,
           only_failed: bool = True
       ) -> DetailedPlanState:
           """从 checkpoint 恢复，只执行失败的维度"""
           state = self.checkpoint_manager.load(checkpoint_id)

           if only_failed:
               failed_dims = state.get("failed_dimensions", [])
               completed = state.get("detailed_dimension_reports", {})

               # 只保留失败维度的状态
               pending = self.get_pending_dimensions(
                   state.get("required_dimensions", []),
                   completed,
                   failed_dims
               )

               state["dimensions_to_execute"] = pending

           return state

   def _calculate_wave_from_dependencies(dimensions: Dict[str, Any]) -> Dict[int, List[str]]:
       """
       根据依赖关系自动计算 Wave 执行顺序

       算法：
       1. 无依赖的维度 → Wave 1
       2. 依赖 Wave N 的维度 → Wave N+1
       3. 迭代直到所有维度分配完毕
       """
       waves = {}
       assigned = set()
       wave_num = 1

       # 按层级分组处理
       for layer in [1, 2, 3]:
           layer_dims = {k: v for k, v in dimensions.items() if v.get("layer") == layer}

           while True:
               # 找出当前 Wave 可执行的维度
               ready = []
               for dim_key, dim_config in layer_dims.items():
                   if dim_key in assigned:
                       continue

                   # 检查同层依赖是否满足
                   deps = dim_config.get("dependencies", {})
                   same_layer_deps = deps.get(f"layer{layer}_{'analyses' if layer == 1 else 'concepts' if layer == 2 else
   'plans'}", [])

                   if all(d in assigned for d in same_layer_deps):
                       ready.append(dim_key)

               if not ready:
                   break

               wave_key = f"layer{layer}_wave{wave_num}"
               if wave_key not in waves:
                   waves[wave_key] = []
               waves[wave_key].extend(ready)
               assigned.update(ready)
               wave_num += 1

           wave_num = 1  # 重置为下一层

       return waves

   def _calculate_wave_for_layer(dimensions: Dict[str, Any], layer: int) -> Dict[int, List[str]]:
       """
       根据同层依赖自动计算 Wave

       Args:
           dimensions: 所有维度配置
           layer: 层级（2 或 3）

       Returns:
           {wave_number: [dimension_keys]}
       """
       # 获取同层依赖字段名
       dep_field = {
           2: "layer2_concepts",
           3: "layer3_plans"
       }.get(layer)

       # 获取该层所有维度
       layer_dims = {k: v for k, v in dimensions.items() if v.get("layer") == layer}

       waves = {}
       assigned = set()
       wave_num = 1

       while len(assigned) < len(layer_dims):
           # 找出当前 Wave 可执行的维度
           ready = []
           for dim_key, dim_config in layer_dims.items():
               if dim_key in assigned:
                   continue

               # 获取同层依赖
               deps = dim_config.get("dependencies", {}).get(dep_field, [])

               # 检查依赖是否都已分配
               if all(d in assigned for d in deps):
                   ready.append(dim_key)

           if not ready:
               break  # 避免死循环

           waves[wave_num] = ready
           assigned.update(ready)
           wave_num += 1

       return waves
